# @continuum/contract

Pure type definitions and constants for the Continuum SDK -- zero runtime dependencies.

This package defines the shared vocabulary that all other Continuum packages depend on. It contains no executable code, only TypeScript interfaces, type aliases, and `as const` constant objects.

## Installation

```bash
npm install @continuum/contract
```

## Types

### SchemaSnapshot

The structural definition of a UI, typically generated by an AI agent.

```typescript
interface SchemaSnapshot {
  schemaId: string;              // stable identifier for the form/schema (e.g. "loan-app")
  version: string;               // increments on each schema push (e.g. "1.0", "2.0")
  components: ComponentDefinition[];
}
```

### ComponentDefinition

A single UI component within a schema.

```typescript
interface ComponentDefinition {
  id: string;                    // unique identifier within this schema version
  type: string;                  // component type (e.g. "input", "select", "toggle")
  key?: string;                  // stable semantic key for matching across versions
  path?: string;                 // display label or path
  hash?: string;                 // schema shape hash for migration detection
  stateType?: string;            // hint about the state shape
  stateShape?: unknown;          // additional metadata (e.g. dropdown options)
  migrations?: MigrationRule[];  // declarative migration rules for hash transitions
  children?: ComponentDefinition[];  // nested child components
}
```

### MigrationRule

Declares how state should be migrated when a component's hash changes.

```typescript
interface MigrationRule {
  fromHash: string;     // hash of the prior schema shape
  toHash: string;       // hash of the new schema shape
  strategyId?: string;  // key into the strategy registry
}
```

### StateSnapshot

The user's accumulated input state applied against a schema.

```typescript
interface StateSnapshot {
  values: Record<string, ComponentState>;  // componentId → state
  meta: StateMeta;
  valuesMeta?: Record<string, ValueMeta>;
}
```

### ComponentState

Union type representing the state of a single component.

```typescript
type ComponentState =
  | ValueInputState
  | ToggleState
  | SelectionState
  | ViewportState
  | Record<string, unknown>;
```

#### ValueInputState

```typescript
interface ValueInputState {
  value: string | number;
  isDirty?: boolean;
}
```

#### ToggleState

```typescript
interface ToggleState {
  checked: boolean;
}
```

#### SelectionState

```typescript
interface SelectionState {
  selectedIds: string[];
}
```

#### ViewportState

```typescript
interface ViewportState {
  scrollX: number;
  scrollY: number;
  expanded?: boolean;
}
```

### StateMeta

Metadata about the state snapshot.

```typescript
interface StateMeta {
  timestamp: number;
  sessionId: string;
  schemaId?: string;
  schemaVersion?: string;
  schemaHash?: string;
  lastInteractionId?: string;
}
```

### ValueMeta

Per-component metadata.

```typescript
interface ValueMeta {
  lastUpdated?: number;
  lastInteractionId?: string;
}
```

### ContinuitySnapshot

Combined schema + state -- the primary data structure flowing through the system.

```typescript
interface ContinuitySnapshot {
  schema: SchemaSnapshot;
  state: StateSnapshot;
}
```

### Interaction

A recorded user interaction event.

```typescript
interface Interaction {
  id: string;
  sessionId: string;
  componentId: string;
  type: string;
  payload: unknown;
  timestamp: number;
  schemaVersion: string;
}
```

### PendingAction

An action submitted by the user that awaits validation.

```typescript
interface PendingAction {
  id: string;
  componentId: string;
  actionType: string;
  payload: unknown;
  createdAt: number;
  schemaVersion: string;
  status: 'pending' | 'validated' | 'stale' | 'cancelled';
}
```

### Checkpoint

A saved snapshot of the session at a point in time, used for rewind.

```typescript
interface Checkpoint {
  id: string;
  sessionId: string;
  snapshot: ContinuitySnapshot;
  eventIndex: number;
  timestamp: number;
}
```
